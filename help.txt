	help			Afficher l'aide de commandes
	exit 			Quitter l'application
	load			Charger un fichier

	@ 			Selectionner un id (@cible [action] [valeur]), une seule 
	. 			Selectionner un enfant (@cible.enfant [action] [valeur])
	% 			Selectionner selon un modulo (@cible1%cible2 [action] [valeur])
	:			Selectionner selon un intervalle (@cible1:cible2 [action] [valeur])

	hide ids / rulers	Cacher les ids / axes
	show ids / rulers	Montrer les ids / axes	

	move			Deplacer un objet (@cible move [valeur] -> deplace a la valeur souhaitee | @cible move [+/-valeur] -> deplace de +/- la valeur)
	rotate			Tourner un objet (@cible move [valeur] -> pivote a la valeur d'angle souhaitee | @cible move [+/-valeur] -> pivote de +/- la valeur)
	
	undo / redo		Precedent / Suivant
